    int def fact(int n) {
        var res : int
        res = 1

        while(n>0) {
            res = res * n
            n = n - 1
        }

        return res
    }

    void def main() {
        var n : int
        print("Enter number: ")
        read(n)
        print("Factorial = ", fact(n))
    }
